10.Min and Max: 

      function arrayMin(arr) {
        min = arr[0];
        for (let i = 0 ; i < arr.length ; i++){
          if (arr[i] < min ){
            min = arr[i];
          }
        }
        return min;
      
   }
      
   function arrayMax(arr) {
        max = arr[0];
        for (let i = 0 ; i < arr.length ; i++){
          if (arr[i] > max ){
            max = arr[i];
          }
        }
        return max;
      
   }
